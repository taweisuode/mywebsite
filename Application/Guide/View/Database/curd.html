使用ppf提供的新增方法


    $insertData = array(
    'movie_name'=>"test",
    'movie_pic' => '111.jpg',
    'movie_url' => 'www.baidu.com',
    'add_time'   => 'November 06,2017',
    'movie_says'=> '很好看很好看的电影，电影画面很炫丽的'
    );
    $return = $this->db->insert($this->_tablename,$insertData);

    $sql = "INSERT INTO Persons (LastName, Address) VALUES ('Wilson', 'Champs-Elysees') ";

查询

  • 使用ppf提供的查询方法

使用ppf提供的查询方法


    <?php
    $select = $this->db->select("oid,user_id,product_id");
    $select  = $this->db->from($this->_tablename);
    $select  = $this->db->where("product_id = '1'");
    $select  = $this->db->orderby("oid","desc");
    $select  = $this->db->limit(1,10);
    $result = $select->fetchAll();

    也可以使用链式操作

    $select = $this->db->select("oid,product_title,dest_name")
    ->from($_tablename)
    ->where(array("product_id"=>"1751939"))
    ->limit(0,10)
    ->orderby("oid","desc");
    $data = $select->fetchAll();
    ?>

使用原生sql语句查询



    $sql = "select * from t_sys_user where user_sn = '".$user_sn."'";
    $result = $this->query($sql);
    return $result;
