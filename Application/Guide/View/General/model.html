模型¶

什么是模型？

模型是专门用来和数据库打交道的 PHP 类,ppf集成封装了大部分的mysql curd语句，我们就可以很方便的进行数据库操作

可以查看以下实例，查看如何进行model文件的编辑

                            
<?php
error_reporting(E_ALL ^ E_NOTICE);
class DirecotryManageModel extends Model
{
    protected $_tablename = 'directory_manage';
    public function directoryData() {
        $select = $this->db->select("*")->from($this->_tablename);
        $result = $select->fetchAll();
        $main_node_arr = array();
        $children_node = array();
        if($result) {
            foreach($result as $key => $val) {
                if($val['level'] == 1) {
                    $main_node_arr[]  = $val;
                }
                $children_node[$val['parent_node']][] = $val;
            }
            foreach($main_node_arr as $key => $val) {
                $main_node_arr[$key]['children_node'] = $children_node[$val['id']];
            }
        }
        return $main_node_arr;
    }
    public function generalDir() {
        $where = array(
            'controller' => 'General'
        );
        $select = $this->db->select("*")->from($this->_tablename);
        $select->where($where);
        $result = $select->fetchAll();
        return $result;
    }
}
?>
  • ppf中提供了简单的基础的链式操作
  • ppf中还支持很多基础的mysql操作，具体可以查看数据库参考的内容
  • controller为方法名

如何在控制器中调用模型中生成的数据？

                            
<?php

    public function indexAction() {
        $directoryModel = new DirecotryManageModel();
        $generalDir = $directoryModel->generalDir();
        $this->view->assign('generalDir',$generalDir);
        $this->view->show();
    }
?>

ppf实现了自动加载类，所以不需要包含类文件，直接new一个model类就可以调起

 $directoryModel = new DirecotryManageModel();

这里需要注意大小写，以及确保对应到相应的Model文件夹下的model文件
